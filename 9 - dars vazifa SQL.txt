SELECT * FROM models WHERE car_count < 5 ORDER BY car_count;

SELECT * FROM models WHERE car_count > 4 AND car_count < 15 ORDER BY car_count;

SELECT b.brand_name, COUNT(m.model_id) AS avtomobil_soni

FROM brands b
LEFT JOIN models m ON b.brand_id = m.brand_id
GROUP BY b.brand_name;

SELECT c.color_name, COUNT(m.model_id) AS avtomobil_soni
FROM colors c
LEFT JOIN models m ON c.color_id = m.color_id
GROUP BY c.color_name;
